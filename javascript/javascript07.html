01. Math.random() : 랜덤값 반환(0~1 사이의 난수 값을 반환) : 반환(숫자)

이 메소드는 의사 난수를 생성하며, 각 호출마다 다른 값을 반환합니다. 

{
var randomNumber = Math.random(); // 0 이상 1 미만의 무작위 부동 소수점 숫자를 반환
console.log(randomNumber);

var randomInteger = Math.floor(Math.random() * 10); // 0 이상 9 이하의 정수를 반환
console.log(randomInteger)
}
